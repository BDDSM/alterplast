Implements OpenConf.IConfigPlugin
Dim WithEvents Conf As OpenConf.Configurator

Private Function IConfigPlugin_Init(ByVal pConfigurator As OpenConf.IConfigurator) As String
    IConfigPlugin_Init = "SimplePlugin"
    Set Configurator = pConfigurator
    Set Conf = pConfigurator
    DoMsgLine "Hello from SimplePlugin", mBlackErr
End Function

Private Sub IConfigPlugin_Done()
    MsgBox "SimplePlugin Done"
    Set Configurator = Nothing
    Set Conf = Nothing
End Sub

Private Function IConfigPlugin_GetPluginCaps(ByVal capNum As OpenConf.PluginCaps) As Variant
    If capNum = capIconNum Then IConfigPlugin_GetPluginCapabilities = 0
    If capNum = capShowMacros Then IConfigPlugin_GetPluginCapabilities = False
    If capNum = capOpenIn Then IConfigPlugin_GetPluginCapabilities = 1
End Function
